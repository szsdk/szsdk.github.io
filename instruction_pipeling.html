#include <stdio.h>
#include <time.h>
#include <math.h>

int main()
{
    clock_t t;
    int a,b;
    t=clock();
    for(int i=0; i<1e5; ++i) {
        a = sin(i)*2.0;
        b = sin(a);
    }
    t=clock()-t;
    printf("Time: %ld\n", t);
    printf("%d %d\n", a, b);

    t=clock();
    for(int i=0; i<1e5; ++i) {
        a = sin(i)*2.0;
        b = sin(i);
    }
    t=clock()-t;
    printf("Time: %ld\n", t);
    printf("%d %d\n", a, b);
    return 0;
}
